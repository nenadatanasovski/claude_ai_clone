```jsx
// Tooltip component with arrow
function Tooltip({ children, text, position = 'top' }) {
  const [isVisible, setIsVisible] = useState(false)
  const timeoutRef = useRef(null)

  const showTooltip = () => {
    timeoutRef.current = setTimeout(() => {
      setIsVisible(true)
    }, 500) // 500ms delay
  }

  const hideTooltip = () => {
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current)
    }
    setIsVisible(false)
  }

  useEffect(() => {
    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current)
      }
    }
  }, [])

  // Position and arrow classes for top/bottom/left/right
  const positionClasses = { /* ... */ }
  const arrowClasses = { /* ... */ }

  return (
    <div
      className="relative inline-block"
      onMouseEnter={showTooltip}
      onMouseLeave={hideTooltip}
      onFocus={showTooltip}
      onBlur={hideTooltip}
    >
      {children}
      {isVisible && text && (
        <div className={`absolute ${positionClasses[position]} z-50 pointer-events-none`} role="tooltip">
          <div className="relative">
            <div className="bg-gray-900 dark:bg-gray-100 text-white dark:text-gray-900
                          text-xs font-medium px-3 py-1.5 rounded-lg shadow-lg whitespace-nowrap">
              {text}
            </div>
            <div className={`absolute ${arrowClasses[position]}`}></div>
          </div>
        </div>
      )}
    </div>
  )
}
```
